# 简介

inquirer.js 是一个通用交互式命令行用户界面的集合。

# 安装

npm install inquirer

# 使用

  1. 导入模块

    const inquirer = require("inquirer");

  2. 声明问题

    // 输入型
let question = {
  type: "input", // 问题类型
  message: "姓名", // 问题的描述
  name: "name", // 储存当前问题回答的变量
  default: "", // 提供一个默认值

  // 验证用户输入，如果不符合规则则重新输入
  validate(value) {
    if (value === "") {
      return true;
    } else {
      return false;
    }
  },
};

// 询问是否，返回值为布尔型
let question = {
  type: "confirm",
  message: "是否毕业",
  name: "isTrue", // true 或 false
  default: "false",
};

// 单选型
let question = {
  type: "list",
  message: "选择一款前端框架",
  name: "data",
  choices: ["React", "Vue", "Angular"],
};

// 多选型
let question = {
  type: "checkbox",
  message: "选择爱好",
  name: "data",
  choices: ["编程", "音乐", "美术"],
};

// 密码
let question = {
  type: "password",
  message: "输入密码",
  name: "password",
};

// 备注
let question = {
  type: "editor",
  message: "添加备注",
  name: "editor",
};

  3. 根据用户的答案执行 then 方法

    inquirer
  .prompt([
    /* Pass your questions in here */
  ])
  .then((answers) => {
    // Use user feedback for... whatever!!
  })
  .catch((error) => {
    console.log(error);
  });
